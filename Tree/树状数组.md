# 树状数组

## 1.单点修改、区间查询

```c++
#include <iostream>
#include <vector>
using namespace std;

int n;
vector<int> a;
vector<int> t;

int lowbit(int n){
    return n & (-n);
}

void add(int x,int k){
    for(;x<=n;x+=lowbit(x)) t[x]+=k;
}

int query(int x){
    int ans=0;
    for(;x;x-=lowbit(x)) ans+=t[x];
    return ans;
}

void init(){
    for(int i=1;i<=n;i++){
       add(i,a[i]);
    }
}

int main(){
    int m;
    scanf("%d %d",&n,&m);
    a.resize(n+1),t.resize(n+1);
    for(int i=1;i<=n;i++)  scanf("%d",&a[i]);
    init();

    int op,x,y;
    for(int i=0;i<m;i++){
        scanf("%d %d %d",&op,&x,&y);
        if(op==1){
            add(x,y);
        }else{
            printf("%d\n",query(y)-query(x-1));
        }
    }

    return 0;
}
```

## 2.区间修改、单点查询

```c++
#include <iostream>
#include <vector>
using namespace std;

int n;
vector<int> a;
vector<int> t;

int lowbit(int n){
    return n & (-n);
}

void add(int x,int k){
    for(;x<=n;x+=lowbit(x)) t[x]+=k;
}

int query(int x){
    int ans=0;
    for(;x;x-=lowbit(x)) ans+=t[x];
    return ans;
}


int main(){
    int m;
    scanf("%d %d",&n,&m);
    a.resize(n+1),t.resize(n+1);
    for(int i=1;i<=n;i++)  scanf("%d",&a[i]);

    int op,x,y,k;
    for(int i=0;i<m;i++){
        scanf("%d",&op);
        if(op==1){
            scanf("%d %d %d",&x,&y,&k);
            add(x,k);
            add(y+1,-k);
        }else{
            scanf("%d",&x);
            printf("%d\n",a[x]+query(x));
        }
    }

    return 0;
}
```

## 3.区间修改、区间查询

```c++
#include <iostream>
#include <vector>
using namespace std;

long long n;
vector<long long> a;
vector<long long> sum;
vector<long long> t1;
vector<long long> t2;

long long lowbit(int n){
    return n & (-n);
}

void add1(int x,int k){
    for(;x<=n;x+=lowbit(x)) t1[x]+=k;
}
void add2(int x,int k){
    for(;x<=n;x+=lowbit(x)) t2[x]+=k;
}

long long query1(int x){
    long long ans=0;
    for(;x;x-=lowbit(x)) ans+=t1[x];
    return ans;
}
long long query2(int x){
    long long ans=0;
    for(;x;x-=lowbit(x)) ans+=t2[x];
    return ans;
}

void init(){
    sum[1]=a[1];
    long long s=a[1];
    for(int i=2;i<=n;i++){
        s+=a[i];
        sum[i]=s;
    }
}


int main(){
    int op;
    long long m,x,y,k;
    scanf("%lld %lld",&n,&m);
    a.resize(n+1),sum.resize(n+1),t1.resize(n+1),t2.resize(n+1);
    for(int i=1;i<=n;i++)  scanf("%lld",&a[i]);
    init();

    for(int i=0;i<m;i++){
        scanf("%d",&op);
        if(op==1){
            scanf("%lld %lld %lld",&x,&y,&k);
            add1(x,k),add1(y+1,-k);
            add2(x,x*k),add2(y+1,-(y+1)*k);
        }else{
            scanf("%lld %lld",&x,&y);
            long long ans1=sum[y]+(y+1)*query1(y)-query2(y);
            long long ans2=sum[x-1]+(x)*query1(x-1)-query2(x-1);
            printf("%lld\n",ans1-ans2);
        }
    }

    return 0;
}
```
